Some ways to generate fibonacci in Haskell

Posted on March 5, 2020 
fib :: Integral a => [a]
fib = 1 : scanl (+) 1 fib

fib' :: Integer -> Integer
fib' n 
  | n < 2    = n
  |otherwise = fib' (n-1) + fib' (n-2)

fibZip :: Integral a => [a]
fibZip = 0 : 1 : zipWith (+) fibZip (tail fibZip)

fibAnamorphism :: Integral a => [a]
fibAnamorphism = unfoldr (\(a,b) -> let fibNum = a+b 
                                    in Just (fibNum, (b,fibNum))) (0,1)

fibAnamorphism' :: Integral a => [a]
fibAnamorphism' = fst <$> iterate (\(a,b) -> (b,a+b)) (0,1)
